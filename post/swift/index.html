继承

  • 面对多次继承后的协议，类必须实现其的全部方法(前提：继承自该协议的类)

protocol V1 {
	func sing()
}

protocol V2 {
	func sing2()
}

protocol V3 : V1,V2{
	func sing3()
}


extension V1 {
	func sing() {
		print("V1的sing")
	}
}

class D3: V3 {

	
	func sing2() {
		
	}
	
	func sing3() {
		
	}
	
	
}

let d3 = D3()

print(d3.sing())
